import React from 'react';
import PropTypes from 'prop-types';

class Total extends React.Component {
   render() {
      const { num1, num2, num3 } = this.props;
      // Calculate the sum of all three numbers
      const sum = parseInt(num1) + parseInt(num2) + parseInt(num3);
      
      return (
         <div>
            The total of {num1}, {num2}, and {num3} is {sum}
         </div>
      );
   }
}

// Assignment must match the Class Name
Total.propTypes = {
   num1: PropTypes.number,
   num2: PropTypes.number,
   num3: PropTypes.number
};

export default Total;

import ReactDOM from 'react-dom';

// Passing the three numbers as props
ReactDOM.render(
   <Total num1={10} num2={20} num3={30} />, 
   document.getElementById('root')
);
